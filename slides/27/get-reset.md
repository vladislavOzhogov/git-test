git reset --soft HEAD~1  # Отменить последний коммит, но сохранить изменения в индексе
git reset 'HEAD@{1}' # Вернуться на последнюю версию
git reset --hard HEAD~1  # Сбросить на состояние удаленной ветки, удалив локальные изменения
git reset HEAD file.txt  # Убрать файл из индекса, не меняя его в директории
