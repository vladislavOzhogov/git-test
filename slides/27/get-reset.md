# Отменить последний коммит, но сохранить изменения в индексе
git reset --soft HEAD~1  

git status

# Вернуться на последнюю версию
git reset 'HEAD@{1}' 

# Сбросить на состояние удаленной ветки, удалив локальные изменения
git reset --hard HEAD~1  

git status

# Убрать файл из индекса, не меняя его в директории
git reset HEAD file.txt
